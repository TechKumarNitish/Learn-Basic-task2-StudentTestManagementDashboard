SELECT 
 SD.roll_number,
 SD.student_name,
 SD.class,
 SD.section,
 CD.subject,
 CD.chapter_name,
 CD.test_name,
 CD.score,
 CD.time_taken_in_sec,
 SD.school_name
FROM 
  public.student_data SD
JOIN 
  public.chapter_test_data CD 
ON 
  SD.roll_number = CD.roll_number
WHERE 
  ({{!school_name.selectedOptionValue}} OR school_name = {{school_name.selectedOptionValue}})
  AND
  ({{!class_name.selectedOptionValue}} OR class = {{class_name.selectedOptionValue}})
  AND
  ({{!section.selectedOptionValue}} OR section = {{section.selectedOptionValue}})
	 AND
  ({{!subject_name.selectedOptionValue}} OR subject = {{subject_name.selectedOptionValue}})
	AND 
	 ({{!Select2.selectedOptionValue}} OR chapter_name = {{Select2.selectedOptionValue}})
	AND 
	 ({{!test_name.selectedOptionValue}} OR test_name = {{test_name.selectedOptionValue}});
